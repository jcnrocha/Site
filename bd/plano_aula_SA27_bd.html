Estrutura básica do SELECT
-- sintaxe completa — ordem de escrita
SELECT    coluna1, coluna2    -- quais colunas retornar
FROM     tabela          -- de qual tabela
WHERE    coluna = valor   -- filtro de linhas
ORDER BY coluna ASC;    -- ordenação (ASC = padrão)

-- exemplo: filmes de ação ordenados por ano
SELECT titulo, ano
FROM filme
WHERE genero = 'Ação'
ORDER BY ano DESC;
Operadores
🔢 Comparação
=
Igual aWHERE ano = 2023
<>
Diferente deWHERE genero <> 'Terror'
> <
Maior / menorWHERE ano > 2020
>= <=
Maior ou igual / menor ou igualWHERE ano <= 2015
🔗 Lógicos
AND
Ambas as condições verdadeirasWHERE genero = 'Ação' AND ano > 2018
OR
Pelo menos uma condição verdadeiraWHERE cidade = 'SP' OR cidade = 'RJ'
NOT
Nega a condiçãoWHERE NOT genero = 'Terror'
✨ Especiais
BETWEEN
Intervalo inclusivo de valoresWHERE ano BETWEEN 2015 AND 2023
LIKE
Padrão em texto — % = qualquer texto, _ = 1 charWHERE nome LIKE 'A%'
IN
Lista de valores possíveisWHERE cidade IN ('Curitiba', 'São Paulo')
Ordem de escrita vs. ordem de execução
✍️ Como você escreve
1SELECT— quais colunas exibir
2FROM— de qual tabela
3WHERE— condição de filtro
4ORDER BY— como ordenar
⚙️ Como o banco executa
1FROM— carrega a tabela
2WHERE— filtra as linhas
3SELECT— seleciona as colunas
4ORDER BY— ordena o resultado
DISTINCT — eliminar duplicatas
-- sem DISTINCT: pode retornar 'Ação' várias vezes
SELECT genero FROM filme;

-- com DISTINCT: cada gênero aparece uma única vez
SELECT DISTINCT genero FROM filme;
